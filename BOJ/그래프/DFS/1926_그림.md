```python
import sys

input = sys.stdin.readline

N, M = map(int, input().split())
graph = []
count = 0
max_visit = 0

for _ in range(N):
  graph.append(list(map(int, input().split())))

dx = [1, 0, -1, 0]
dy = [0, 1, 0, -1]

def dfs_stack(start_x, start_y):
  visit_count = 1
  stack = list()
  stack.append((start_x, start_y))
  graph[start_x][start_y] = -1

  while stack:
    x, y = stack.pop()

    for i in range(4):
      nx = x + dx[i]
      ny = y + dy[i]
      if nx > N - 1 or nx < 0 or ny > M - 1 or ny < 0:
        continue
      if graph[nx][ny] == 1:
        graph[nx][ny] = -1
        visit_count += 1
        stack.append((nx, ny))

  return visit_count


for i in range(N):
  for j in range(M):
    if graph[i][j] == 1:
      max_visit = max(max_visit, dfs_stack(i, j))
      count += 1

print(count)
print(max_visit)
```

#### 재귀 dfs를 이용한 풀이

메모리 초과 났다.

```python
def dfs(x, y, visit_count):
  graph[x][y] = -1

  for i in range(4):
    nx = x + dx[i]
    ny = y + dy[i]
    if nx > N - 1 or nx < 0 or ny > M - 1 or ny < 0:
      continue
    if graph[nx][ny] == 1:
      visit_count = dfs(nx, ny, visit_count + 1)
  return visit_count

for i in range(N):
  for j in range(M):
    if graph[i][j] == 1:
      max_visit = max(max_visit, dfs(i, j, 1))
      count += 1
```
